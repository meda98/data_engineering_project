# First time or after code/Dockerfile changes
docker compose up --build

# Subsequent runs with no image changes
docker compose up

# Delete sensor_readings table
docker compose up -d postgres
docker exec -it postgres psql -U de_user -d sensordb -c "DROP TABLE IF EXISTS sensor_readings;"

# Remove all Kafka messages (reset the topic)
docker compose up -d kafka
docker exec -it kafka /opt/kafka/bin/kafka-topics.sh \
  --bootstrap-server kafka:9093 \
  --delete \
  --topic sensor_readings
